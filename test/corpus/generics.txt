    data: T,
}

fn identity<T>(x: T) T {
    x
}

pub fn main() void {
    let x = identity::<usize>(42);
    let y = identity(Foo { data: 42 });
    let z: Foo::<Foo::<usize> > = Foo::<Foo::<usize> > { data: identity(Foo::<usize> { data: 42 }) };
}
